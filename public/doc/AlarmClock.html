def play_sound(file_name)
  if RUBY_PLATFORM.match(/mingw|mswin/)
    Sound.play Home::Engine.root.join('app', 'assets', 'audio', file_name).to_s
  else
    pid = fork{ exec 'mpg123','-q', file_name }
  end
end
