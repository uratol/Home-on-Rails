Overview

Драйвер обеспечивает прямой доступ к портам ввода-вывода GPIO (general-purpose input/output). В адресе указывается номер пина по BCM-нумерации При инициализации порта, указанного в адресе, устанавливает подтяжку по напряжению (pull up)

def self.watch(&block)

  pins = sensors.uniq.pluck(:address).map(&:to_i)
  
  @threads.each(&:kill) if (@threads ||= []).any?

  puts "GPIO pins #{ pins } will be watching"
  
  pins.each do |unmapped_pin|
    @threads << Thread.new(block) do |trigger|
      mapped_pin = map_pin(unmapped_pin)
      value = nil
      puts "loop pin #{ unmapped_pin }(#{ mapped_pin })"
      loop do
        last_value = value
        value = io.digital_read(mapped_pin)
        if value != last_value
          trigger.call(unmapped_pin, value)
        end
        sleep(0.01)
      end
    end
  end
  @threads.each(&:join)
end
