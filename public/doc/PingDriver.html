def self.extended(base)
  ip_block = /\d{,2}|1\d{2}|2[0-4]\d|25[0-5]/
  ip_mask = /\A#{ip_block}\.#{ip_block}\.#{ip_block}\.#{ip_block}\z/
  base.singleton_class.validates :address, presence: true, format: { with: ip_mask }
  base.binary = true
end
