  def call event_name, options = {}
#puts "events#call event_name=#{event_name}, options=#{options}, length=#{inspect}"
    select{|e| e.name==event_name && e.options[:source_row]==options[:source_row]}.each do |event|
      params = options[:params]
      if params
        event.call *params
      else
        event.call
      end    
    end  
  end
