def between?(from, to)
  if TimeRangeComparable.args_is_time_str(from ,to)
    from = from.in_time_zone
    to = to.in_time_zone

    if from > to
      if self > from
        to += 1.day
      else
        from -= 1.day
      end
    end
  end
  super
end
